/* Generate comprehensive banking dataset */
data MODELS.credit_training;
    call streaminit(&RANDOM_SEED);
    
    do CustomerID = 100001 to (100000 + &NUM_RECORDS);
        /* Core demographic features */
        Age = 22 + floor(rand('UNIFORM') * 48);
        EmploymentYears = min(Age - 18, floor(rand('UNIFORM') * (Age - 21)));
        
        /* Income with realistic age correlation */
        base_income = 35000 + (Age - 22) * 1500 + EmploymentYears * 2000;
        Income = max(25000, base_income + rand('NORMAL') * 18000);
        
        /* Credit Score influenced by income and employment */
        credit_base = 580 + (Income/100000) * 120 + EmploymentYears * 4;
        CreditScore = max(300, min(850, floor(credit_base + rand('NORMAL') * 50)));
        
        /* Alternative Data: Social Media Risk Score */
        social_risk_factor = (1 - (CreditScore - 300)/550) * 0.4 + 
                           DebtToIncome * 0.3 + rand('UNIFORM') * 0.3;
        if social_risk_factor < 0.2 then SocialMediaRiskScore = 1;
        else if social_risk_factor < 0.4 then SocialMediaRiskScore = 2;
        else if social_risk_factor < 0.6 then SocialMediaRiskScore = 3;
        else if social_risk_factor < 0.8 then SocialMediaRiskScore = 4;
        else SocialMediaRiskScore = 5;
        
        output;
    end;
run;

build_credit_model.sas

/* Advanced logistic regression model */
proc logistic data=work.train_set;
    title "Enterprise Credit Risk Model v&MODEL_VERSION";
    
    class Region(ref='Urban') LoanType(ref='Personal') 
          CustomerSegment(ref='Basic') AgeRisk(ref='Prime') 
          EmpStability(ref='Stable') / param=ref;
          
    model DefaultRisk(event='1') = 
        /* Traditional credit variables */
        CreditScore DebtToIncome Income EmploymentYears Age LoanAmount
        
        /* Alternative data features */
        SocialMediaRiskScore DeviceUsageScore
        
        /* Customer relationship variables */
        NumberOfProducts HasInvestmentAccount AccountTenure
        
        /* Behavioral indicators */
        TransactionFrequency AverageBalance LatePayments OverdraftEvents
        
        /* Derived features */
        CreditUtilization DebtScore_Interaction Income_Credit_Ratio
        
        /* Categorical variables */
        Region LoanType CustomerSegment AgeRisk EmpStability
        
        / selection=stepwise slentry=0.1 slstay=0.05 details;
        
    /* Generate performance metrics */
    roc; 
    output out=work.model_output p=predicted_prob;
run;

def calculate_comprehensive_assessment(self, data):
    """Advanced risk calculation with banking coefficients"""
    
    # Extract features
    credit_score = data.get('creditScore', 650)
    income = data.get('income', 50000)
    debt_ratio = data.get('debtToIncomeRatio', 0.3)
    social_risk = data.get('socialMediaRiskScore', 3)
    
    # Use realistic banking coefficients
    linear_combo = -2.8
    linear_combo += (850 - credit_score) * 0.0045  # Credit impact
    linear_combo -= (income - 50000) * 0.000012   # Income protection
    linear_combo += debt_ratio * 4.2              # Major risk factor
    linear_combo += (social_risk - 1) * 0.3       # Alt data impact
    
    # Apply logistic transformation
    risk_probability = 1 / (1 + math.exp(-linear_combo))
    
    # Determine risk level with business rules
    if risk_probability < 0.25:
        risk_level = 'low_risk'
    elif risk_probability < 0.6:
        risk_level = 'medium_risk'
    else:
        risk_level = 'high_risk'
    
    return {
        'risk_probability': risk_probability,
        'risk_level': risk_level,
        'risk_adjusted_pricing': self._calculate_pricing(risk_probability),
        'customer_segment': self._determine_segment(data),
        'upsell_products': self._recommend_products(data, risk_level)
    }

app = FastAPI(
    title="SAS Banking Intelligence API",
    description="Enterprise credit risk assessment with alternative data",
    version="2.1"
)

# Enable CORS for mobile access
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

class CreditAssessmentRequest(BaseModel):
    creditScore: int = Field(ge=300, le=850)
    income: float = Field(gt=0)
    debtToIncomeRatio: float = Field(ge=0, le=1)
    socialMediaRiskScore: int = Field(ge=1, le=5)
    # ... additional fields

@app.post("/assess/comprehensive")
async def assess_credit_risk(request: CreditAssessmentRequest):
    """Comprehensive credit risk assessment endpoint"""
    try:
        assessment = sas_model.calculate_comprehensive_assessment(
            request.dict()
        )
        return assessment
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
